# Типичные ошибки с заголовками

## 1. Отсутствие include guards

```c
// ПЛОХО
struct Data { int x; };

// main.c
#include "header.h"
#include "other.h"  // включает header.h
// ОШИБКА: повторное определение!

// ХОРОШО
#ifndef HEADER_H
#define HEADER_H
struct Data { int x; };
#endif
```

## 2. Лишние зависимости

```c
// header.h — ПЛОХО
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct { } Data;  // не использует ничего из включений

// ХОРОШО
typedef struct { } Data;  // без лишних включений
```

## 3. Круговые зависимости

```c
// a.h
#include "b.h"
typedef struct { B* b; } A;

// b.h
#include "a.h"  // ОШИБКА: цикл!
typedef struct { A* a; } B;

// РЕШЕНИЕ: forward declaration
// b.h
struct A;  // forward
typedef struct { struct A* a; } B;
```

## 4. Определение переменных в заголовке

```c
// header.h — ПЛОХО
int global_counter = 0;  // определение!

// При включении в 2 файла — ошибка линковки

// ХОРОШО
extern int global_counter;  // объявление

// В .c файле
int global_counter = 0;  // определение
```

## 5. Определение функций в заголовке

```c
// header.h — ПЛОХО
void helper() {  // определение!
    // код
}

// При включении в 2 файла — ошибка линковки

// ХОРОШО
// Вариант 1: extern + определение в .c
void helper(void);  // объявление

// Вариант 2: static inline
static inline void helper(void) {
    // код
}
```

## 6. Missing stdlib.h для size_t

```c
// header.h — ПЛОХО
void process(size_t size);  // size_t не определён!

// ХОРОШО
#include <stddef.h>
void process(size_t size);
```

## 7. Неправильный порядок включений

```c
// ПЛОХО
#include "mylib.h"
#include <stdio.h>  // системные после своих
#include "other.h"

// ХОРОШО
#include <stdio.h>   // системные
#include <stdlib.h>
#include "mylib.h"   // свои
#include "other.h"
```

## 8. Макросы конфликтуют

```c
// header.h — ПЛОХО
#define MAX 100
#define MIN 0

// Пользователь использует другие MAX/MIN
#include <algorithm>  // конфликт!

// ХОРОШО
#define MYLIB_MAX 100
#define MYLIB_MIN 0
```

## 9. Утечка макросов

```c
// header.h — ПЛОХО
#define DEBUG_PRINT(x) printf(x)

// Макрос остаётся после включения
#include "header.h"
DEBUG_PRINT("test");  // работает везде

// ХОРОШО
#ifdef MYLIB_DEBUG
#define DEBUG_PRINT(x) printf(x)
#else
#define DEBUG_PRINT(x)
#endif

// Или undef в конце
#define TEMP_MACRO 1
// использование
#undef TEMP_MACRO
```

## 10. Не тот путь включения

```c
// ПЛОХО
#include "../src/mylib.h"  // относительный путь
#include "C:\headers\lib.h"  // абсолютный путь

// ХОРОШО
#include "mylib.h"  // компилятор найдёт через -I
```

```bash
gcc -I./include main.c
```
