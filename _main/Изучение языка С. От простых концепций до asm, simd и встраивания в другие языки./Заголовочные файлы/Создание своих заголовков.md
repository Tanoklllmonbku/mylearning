# Создание своих заголовков

Правильное создание заголовочных файлов — ключ к поддерживаемому коду.

## Структура заголовка

```c
// mylib.h
#ifndef MYLIB_H
#define MYLIB_H

// 1. Подключение зависимостей
#include <stddef.h>
#include <stdint.h>

// 2. C++ совместимость
#ifdef __cplusplus
extern "C" {
#endif

// 3. Версия
#define MYLIB_VERSION_MAJOR 1
#define MYLIB_VERSION_MINOR 0

// 4. Объявления типов
typedef struct {
    int x;
    int y;
} Point;

// 5. Макросы
#define POINT_INIT(x, y) {(x), (y)}

// 6. Прототипы функций
Point* point_create(int x, int y);
void point_destroy(Point* p);
int point_distance(const Point* a, const Point* b);

// 7. Inline функции
static inline int point_x(const Point* p) {
    return p->x;
}

// 8. C++ совместимость
#ifdef __cplusplus
}
#endif

#endif // MYLIB_H
```

## Реализация

```c
// mylib.c
#include "mylib.h"
#include <stdlib.h>
#include <math.h>

Point* point_create(int x, int y) {
    Point* p = malloc(sizeof(Point));
    if (p) {
        p->x = x;
        p->y = y;
    }
    return p;
}

void point_destroy(Point* p) {
    free(p);
}

int point_distance(const Point* a, const Point* b) {
    int dx = a->x - b->x;
    int dy = a->y - b->y;
    return (int)sqrt(dx*dx + dy*dy);
}
```

## Использование

```c
// main.c
#include "mylib.h"
#include <stdio.h>

int main() {
    Point* p1 = point_create(0, 0);
    Point* p2 = point_create(3, 4);
    
    int dist = point_distance(p1, p2);
    printf("Distance: %d\n", dist);
    
    point_destroy(p1);
    point_destroy(p2);
    
    return 0;
}
```

## Компиляция

```bash
gcc -c mylib.c -o mylib.o
gcc -c main.c -o main.o
gcc mylib.o main.o -o program -lm
```

## Лучшие практики

### 1. Именование

```c
// Префикс для всех символов
#define MYLIB_MAX_SIZE 100
typedef struct mylib_data mylib_data_t;
void mylib_init(mylib_data_t* data);
```

### 2. Минимальные зависимости

```c
// ПЛОХО — лишние зависимости
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

// ХОРОШО — только необходимое
#include <stddef.h>  // только size_t
```

### 3. Forward declarations

```c
// Вместо #include "other.h"
struct OtherType;  // forward declaration

void process(struct OtherType* obj);
```

### 4. Документирование

```c
/**
 * Создать новую точку.
 * @param x Координата X
 * @param y Координата Y
 * @return Указатель на точку или NULL при ошибке
 */
Point* point_create(int x, int y);
```
