# Структура заголовка

Правильная структура заголовочного файла обеспечивает надёжность и удобство использования.

## Типовая структура

```c
// 1. Include guards
#ifndef MYLIB_H
#define MYLIB_H

// 2. Подключение необходимых заголовков
#include <stddef.h>
#include <stdint.h>

// 3. C++ совместимость
#ifdef __cplusplus
extern "C" {
#endif

// 4. Объявления типов
typedef struct {
    int x;
    int y;
} Point;

// 5. Объявления констант и макросов
#define MAX_SIZE 100
#define VERSION "1.0.0"

// 6. Прототипы функций
void init(void);
int process(int value);
Point* create_point(int x, int y);
void destroy_point(Point* p);

// 7. Inline функции (опционально)
static inline int square(int x) {
    return x * x;
}

// 8. C++ совместимость
#ifdef __cplusplus
}
#endif

#endif // MYLIB_H
```

## Элементы

### 1. Include guards

```c
#ifndef HEADER_H
#define HEADER_H
// содержимое
#endif
```

### 2. Зависимости

```c
#include <stdio.h>    // системные сначала
#include <stdlib.h>
#include "mylib.h"    // свои потом
```

### 3. C++ совместимость

```c
#ifdef __cplusplus
extern "C" {
#endif

void c_function();

#ifdef __cplusplus
}
#endif
```
