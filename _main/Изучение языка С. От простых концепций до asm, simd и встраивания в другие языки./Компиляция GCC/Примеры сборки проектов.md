# Примеры сборки проектов

Практические примеры сборки различных проектов.

## Проект 1: Один файл

```
project/
└── main.c
```

```bash
# Компиляция
gcc -Wall -O2 main.c -o program

# Makefile
all: program

program: main.c
	gcc -Wall -O2 $< -o $@

clean:
	rm -f program
```

## Проект 2: Несколько файлов

```
project/
├── main.c
├── utils.c
├── utils.h
└── Makefile
```

```c
// utils.h
#ifndef UTILS_H
#define UTILS_H
int add(int a, int b);
void print_result(int result);
#endif
```

```c
// utils.c
#include "utils.h"
#include <stdio.h>

int add(int a, int b) { return a + b; }

void print_result(int result) {
    printf("Result: %d\n", result);
}
```

```c
// main.c
#include "utils.h"

int main() {
    int result = add(2, 3);
    print_result(result);
    return 0;
}
```

```bash
# Компиляция
gcc -Wall -O2 main.c utils.c -o program

# Makefile
CC = gcc
CFLAGS = -Wall -O2
OBJS = main.o utils.o

program: $(OBJS)
	$(CC) $(CFLAGS) $(OBJS) -o $@

%.o: %.c utils.h
	$(CC) $(CFLAGS) -c $<

clean:
	rm -f $(OBJS) program
```

## Проект 3: С библиотекой

```
project/
├── src/
│   ├── main.c
│   └── lib/
│       ├── mylib.c
│       └── mylib.h
└── Makefile
```

```bash
# Makefile
CC = gcc
CFLAGS = -Wall -O2
INCLUDE = -I./src/lib

# Статическая библиотека
LIB_SRC = src/lib/mylib.c
LIB_OBJ = $(LIB_SRC:.c=.o)
LIB_NAME = libmylib.a

# Основной файл
MAIN_SRC = src/main.c
MAIN_OBJ = $(MAIN_SRC:.c=.o)

all: program

# Создание библиотеки
$(LIB_NAME): $(LIB_OBJ)
	ar rcs $@ $^

# Компиляция библиотеки
src/lib/%.o: src/lib/%.c
	$(CC) $(CFLAGS) $(INCLUDE) -c $< -o $@

# Компиляция main
src/%.o: src/%.c
	$(CC) $(CFLAGS) $(INCLUDE) -c $< -o $@

# Линковка
program: $(MAIN_OBJ) $(LIB_NAME)
	$(CC) $(CFLAGS) $(MAIN_OBJ) -L. -lmylib -o $@

clean:
	rm -f $(LIB_OBJ) $(MAIN_OBJ) $(LIB_NAME) program
```

## Проект 4: С математикой

```c
// math_example.c
#include <stdio.h>
#include <math.h>

int main() {
    double x = sqrt(2.0);
    printf("sqrt(2) = %f\n", x);
    return 0;
}
```

```bash
# Компиляция с math library
gcc -Wall -O2 math_example.c -o program -lm

# Makefile
program: math_example.c
	gcc -Wall -O2 $< -o $@ -lm
```

## Проект 5: С pthread

```c
// thread_example.c
#include <stdio.h>
#include <pthread.h>

void* worker(void* arg) {
    printf("Thread working\n");
    return NULL;
}

int main() {
    pthread_t thread;
    pthread_create(&thread, NULL, worker, NULL);
    pthread_join(thread, NULL);
    return 0;
}
```

```bash
# Компиляция с pthread
gcc -Wall -O2 thread_example.c -o program -lpthread

# Makefile
program: thread_example.c
	gcc -Wall -O2 $< -o $@ -lpthread
```

## Проект 6: Полный проект

```
project/
├── src/
│   ├── main.c
│   ├── utils.c
│   └── math_ops.c
├── include/
│   ├── utils.h
│   └── math_ops.h
├── lib/
│   └── libexternal.a
├── build/
└── Makefile
```

```makefile
CC = gcc
CFLAGS = -Wall -Wextra -O2 -I./include
LDFLAGS = -L./lib -lexternal -lm

SRCS = $(wildcard src/*.c)
OBJS = $(SRCS:src/%.c=build/%.o)
TARGET = build/program

.PHONY: all clean rebuild

all: $(TARGET)

$(TARGET): $(OBJS)
	$(CC) $(OBJS) $(LDFLAGS) -o $@

build/%.o: src/%.c
	@mkdir -p build
	$(CC) $(CFLAGS) -c $< -o $@

clean:
	rm -rf build

rebuild: clean all
```

## Запуск сборки

```bash
make          # сборка
make clean    # очистка
make rebuild  # пересборка
./build/program  # запуск
```

## Пример из практики

Готовый проект в [[Калькулятор]]:

```bash
cd _resources/base_level_C/calculator

# Простая сборка
gcc -Wall -O2 *.c -o calculator

# Запуск
./calculator

# Примеры ввода
Enter the expression: 2 + 2 * 2
Result: 6

Enter the expression: (2 + 3) * 4
Result: 20
```

Этот проект демонстрирует:
- Многофайловую компиляцию
- Использование заголовочных файлов с include guards
- Разделение логики по модулям (sum, sub, mul, div)
- Парсинг и вычисление выражений
